首页笔记无线AC

# 开启无线功能

AC(config)#wireless
AC(config-wireless)#enable

# 配置静态无线 ip 地址

AC(config-wireless)#static-ip 10.10.2.1

# 创建 ssid

AC(config-wireless)#network 14
AC(config-network)#ssid Wifi-2.4G // 将 ssid 配置为 Wifi-2.4G
AC (config-network)#vlan 140 // 使用 vlan140
AC (config-network)#security mode wpa-personal // 加密方式为 wpa-personal
AC(config-network)#wpa key Wifi-2.4G // 口令为 Wifi-2.4G

AC (config-network)#security mode static-wep // 加密方式为 static-wep
AC (config-network)#wep key type ascii // 配置 share key 的编码类型
AC (config-network)#wep key 1234567890123 // 口令为 1234567890123

AC(config-wireless)#network 1
AC (config-network)#hide ssid // 隐藏 network 1

# 射频管理

AC(config-wireless)#ap profile 1
AC (config-ap)#name WIFI // 将 ap 命名为 WIFI
AC (config-ap)#radio 1 //radio 1 对应 AP 上 2.4Ghz 工作频段，radio 2 对应 AP 上 5Ghz 工作频段
AC (config-ap)#mode bg // 设置射频工作模式为 802.11bg
AC (config-ap)#vap 13 //network1 对应 vap0，network2 对应 vap1。所以这应该是 vap13
AC(config-ap)#enable

AC(config-ap)#radio 2
AC(config-ap)#mode a //802.11a 标准工作在 5GHz 频段
AC (config-ap)#vap 14
AC(config-ap)#enable

# arp 抑制

AC (config-network)#arp-suppression // 开启 arp 抑制

# 用户隔离

AC (config-network)#station-isolation // 开启用户隔离

# 限时策略

基于 network

AC(config-wireless)# network 2
AC(config-network)# time-limit from 7:0 to 17:0 weekday Monday

基于 radio

//2.4G 信号周一到周三早上 0 点到 6 点禁止终端接入

AC(config-wireless)#ap profile 1
AC(config-ap-profile)#radio 1
AC(config-ap-profile-radio)#time-limit from 00:00 to 06:00 weekday monday
AC(config-ap-profile-radio)#time-limit from 00:00 to 06:00 weekday tuesday
AC(config-ap-profile-radio)#time-limit from 00:00 to 06:00 weekday wednesday

# Client QoS

AC (config-wireless)#ap client-qos // 启动 Client Qos 功能
AC (config-wireless)#network 14
AC (config-network)#client-qos bandwidth-limit up 2048 // 配置上行速度为 2Mbps
AC (config-network)#client-qos bandwidth-limit down 5120 // 配置下行速度为 5Mbps

# MAC 黑名单

AC (config-wireless)#mac-authentication-mode black-list // 开启黑名单 (white-list 白名单)
AC (config-wireless)#know-client 00-e1-3a-10-be-51 action global-action // 添加进黑名单，只有 action 设置为 global 时设置才启作用，在黑名单中被拒绝，相反在白名单中被接受。
AC(config-wireless)# network 14
AC (config-network)# mac authentication local // 打开 mac 认证，并设置本地认证

# 无线信号强度低于 50% 的禁止连接

AC(config-wireless)#ap profile 1
AC(config-ap-profile)#radio 1
AC(config-ap-profile-radio)#client-reject rssi-threshold 50

# 时空公平

AC (config-ap-profile-radio)#schedule-mode fair // 每个客户端机会均等获得传输机会

# AP FLOOD

AC (config-wireless)#wireless ap anti-flood interval 10 // 检测时间 10 分钟
AC (config-wireless)#wireless ap anti-flood max-conn-count 5 // 设置最大连接次数 5 分钟
AC (config-wireless)#wireless ap anti-flood agetime 120 // 设置老化时间 2 小时

# 自动信道调整

AC(config-ap-profile)#channel-plan an time 07:00
AC(config-ap-profile)#channel-plan bgn time 07:00
